# Comenzi rapide — YOLO FPGA (Nexys 4 DDR)

## Instalare
pip3 install -r requirements.txt
bash scripts/download_yolo_model.sh
bash scripts/download_coco_scenarios.sh   # 15 poze/scenariu din COCO
bash scripts/setup_scenario_images.sh       # sample mic manual

## Rulare generală
python3 run_detection.py --images datasets/test_images/
python3 run_detection.py --benchmark --images datasets/test_images/

python3 run_detection.py --use-fpga --benchmark
